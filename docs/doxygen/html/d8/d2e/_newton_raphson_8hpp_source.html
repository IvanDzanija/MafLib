98 T x_new = x - (f_x / f_prime_x);
99 T denom = std::abs(x);

101 denom = std::numeric_limits<T>::epsilon();

103 T rel_error = std::abs(x_new - x) / denom;
104 if (rel_error < tolerance) {

108 }
