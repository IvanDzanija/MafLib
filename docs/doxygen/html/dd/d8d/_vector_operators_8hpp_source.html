297template <Numeric T>
298template <Numeric U>
299auto Vector<T>::operator*(const Matrix<U>& other) const {
300 using R = std::common_type_t<T, U>;
301
302 size_t n = size();
303 size_t m = other.row_count();
304 size_t r = other.column_count();
305
306 if (_orientation == COLUMN) {
307 throw std::invalid_argument(
308 "Invalid multiplication: column Vector * Matrix. "
309 "Did you mean Matrix * Vector?");
310 }
311 if (n != m) {
312 throw std::invalid_argument("Dimensions do not match!");
313 }
314
315 Vector<R> result(r, std::vector<R>(r), ROW);
316 for (size_t i = 0; i < r; ++i) {
317 for (size_t j = 0; j < n; ++j) {
318 result.at(i) += at(j) * other.at(j, i);
319 }
320 }
321 return result;
322}
